Polars cheat sheet

General

Install

pip install polars

Import

import polars as pl

Creating/reading DataFrames

# Create DataFrame
df = pl.DataFrame(
  {
    "nrs": [1, 2, 3, None, 5],
    "names": ["foo", "ham", "spam", "egg", None],
    "random": [0.3, 0.7, 0.1, 0.9, 0.6],
    "groups": ["A", "A", "B", "C", "B"],
  }
)

# Read CSV
df = pl.read_csv("https://j.mp/iriscsv",
                 has_header=True)

# Read parquet
df = pl.read_parquet("path.parquet")

Querying data

# Select columns
df.select(["nrs", "names"])

# Filter
df.filter(pl.col("groups") == "B")
